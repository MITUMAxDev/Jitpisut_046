ตัวอย่างการใช้งานภาษา Lua

Lua เป็นภาษาที่ใช้งานง่าย เหมาะสำหรับผู้เริ่มต้นและผู้ที่ต้องการสร้างสคริปต์อย่างรวดเร็ว นี่คือตัวอย่างต่างๆ ของการใช้งานภาษา Lua:

ตัวอย่างที่ 1: Hello World

-- ตัวอย่างการใช้ print                    
print("Hello World")

ตัวอย่างที่ 2: การใช้ตัวแปร

-- การประกาศตัวแปรและการใช้
local x = 10
local y = 20
local sum = x + y
print("ผลรวมของ x และ y คือ " .. sum)

ตัวอย่างที่ 3: การใช้เงื่อนไข

-- การใช้ if-else
local score = 85

if score >= 90 then
  print("A")
elseif score >= 80 then
  print("B")
else
  print("C")
end

ตัวอย่างที่ 4: การใช้ loop

-- การใช้ for-loop
for i = 1, 5 do
  print("ค่า i: " .. i)
end

ตัวอย่างที่ 5: การใช้ฟังก์ชัน

-- การสร้างฟังก์ชันใน Lua
function greet(name)
  print("สวัสดี, " .. name)
end

greet("Lua Programmer")

ตัวอย่างที่ 6: การใช้ตาราง (table)

-- การสร้างและใช้งานตาราง
local fruits = {"apple", "banana", "cherry"}
    
for i = 1, #fruits do
  print(fruits[i])
end
